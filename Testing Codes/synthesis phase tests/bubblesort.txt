void bubblesort(int array[], int length)
{
    int k;
    int j;
    int aux;

    k = length - 1;
    while (k > 0)
    {
        j = 0;
        while (j < k)
        {
            if (array[j] > array[j + 1])
            {
                aux = array[j];
                array[j] = array[j + 1];
                array[j + 1] = aux;
            }
            j = j + 1;
        }
        k = k - 1;
    }
}
